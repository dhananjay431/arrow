let globalDraw;

function createLeaderLine({
  start,
  end,
  color = '#000',
  width = 2,
  dasharray = null,
  pathType = 'straight', // 'straight' | 'curve'
  arrow = false
}) {
  const startEl = document.querySelector(start);
  const endEl = document.querySelector(end);

  if (!startEl || !endEl) {
    console.error('Invalid selectors');
    return;
  }

  // Create SVG root once
  if (!globalDraw) {
    globalDraw = SVG().addTo('body').size('100%', '100%').css({
      position: 'fixed',
      top: 0,
      left: 0,
      pointerEvents: 'none',
      zIndex: 9999
    });
  }

  const line = globalDraw.path().fill('none').stroke({
    color,
    width,
    dasharray
  });

  // Arrow marker
  let marker;
  if (arrow) {
    marker = globalDraw.marker(10, 10, function(add) {
      add.path('M0,0 L10,5 L0,10 Z').fill(color);
    });
    line.marker('end', marker);
  }
  function getAutoSocket(fromRect, toRect) {
  const fromCX = fromRect.left + fromRect.width / 2;
  const fromCY = fromRect.top + fromRect.height / 2;

  const toCX = toRect.left + toRect.width / 2;
  const toCY = toRect.top + toRect.height / 2;

  const dx = toCX - fromCX;
  const dy = toCY - fromCY;

  const absDx = Math.abs(dx);
  const absDy = Math.abs(dy);

  if (absDx > absDy) {
    return dx > 0 ? 'right' : 'left';
  } else {
    return dy > 0 ? 'bottom' : 'top';
  }
}
function getEdgePoint(rect, targetX, targetY) {
  const cx = rect.left + rect.width / 2;
  const cy = rect.top + rect.height / 2;

  const dx = targetX - cx;
  const dy = targetY - cy;

  const absDx = Math.abs(dx);
  const absDy = Math.abs(dy);

  // Decide whether horizontal or vertical edge
  if (absDx > absDy) {
    // LEFT or RIGHT
    if (dx > 0) {
      return {
        x: rect.right,
        y: cy
      };
    } else {
      return {
        x: rect.left,
        y: cy
      };
    }
  } else {
    // TOP or BOTTOM
    if (dy > 0) {
      return {
        x: cx,
        y: rect.bottom
      };
    } else {
      return {
        x: cx,
        y: rect.top
      };
    }
  }
}
function getSocketPoint(rect, socket) {
  const cx = rect.left + rect.width / 2;
  const cy = rect.top + rect.height / 2;

  switch (socket) {
    case 'top':
      return { x: cx, y: rect.top };
    case 'bottom':
      return { x: cx, y: rect.bottom };
    case 'left':
      return { x: rect.left, y: cy };
    case 'right':
      return { x: rect.right, y: cy };
  }
}
function update() {
  const startRect = startEl.getBoundingClientRect();
  const endRect = endEl.getBoundingClientRect();

  // Auto decide sockets
  const startSocket = getAutoSocket(startRect, endRect);
  const endSocket = getAutoSocket(endRect, startRect);

  const startPoint = getSocketPoint(startRect, startSocket);
  const endPoint = getSocketPoint(endRect, endSocket);

  let d;

  if (pathType === 'curve') {
    const curveOffset = 60;

    d = `M ${startPoint.x} ${startPoint.y}
         C ${
           startSocket === 'left' ? startPoint.x - curveOffset :
           startSocket === 'right' ? startPoint.x + curveOffset :
           startPoint.x
         } ${
           startSocket === 'top' ? startPoint.y - curveOffset :
           startSocket === 'bottom' ? startPoint.y + curveOffset :
           startPoint.y
         },
         ${
           endSocket === 'left' ? endPoint.x - curveOffset :
           endSocket === 'right' ? endPoint.x + curveOffset :
           endPoint.x
         } ${
           endSocket === 'top' ? endPoint.y - curveOffset :
           endSocket === 'bottom' ? endPoint.y + curveOffset :
           endPoint.y
         },
         ${endPoint.x} ${endPoint.y}`;
  } else {
    d = `M ${startPoint.x} ${startPoint.y} L ${endPoint.x} ${endPoint.y}`;
  }

  line.plot(d);
}

  // Initial draw
  update();

  // Auto update
  window.addEventListener('resize', update);
  window.addEventListener('scroll', update);

  return {
    update,
    remove: () => {
      line.remove();
      window.removeEventListener('resize', update);
      window.removeEventListener('scroll', update);
    }
  };
}

  pathType: 'curve',
